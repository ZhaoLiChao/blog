    # 2. createStore.js ==> 创建store

    export default function createStore (reducer,initialState){
    let state = initialState;
    let callbacks = [];

    function getState(){        //获取当前的状态
        return state;
    }

    function dispatch(action){
        state = reducer(state, action);
        callbacks.forEach(callFn => callFn());
        return action;          //将action返回，以便在组件中做其他操作
    }

    // 注意：dispatch需要自己调用一次，因为有可能调用createStore时，没有传初始化的state，需要去reducer中初始化，但并不需要去修改state
    dispatch({ type: '@@REDUX_INIT' });         //不会执行reducer里面修改state的逻辑，只会初始化state

    function subscribe(callback){
        callbacks.push(callback);
        // 返回一个方法，调用它后取消订阅
        // 即在组件中，订阅需要在组件销毁时取消，否则会造成内存泄露。
        return function () {        
            let index = callbacks.indexOf(callback);
            callbacks.splice(index, 1);
        }
    }

    return {
        getState,
        dispatch,
        subscribe
    }
}

    # 3. bindActionCreators.js ==> 自动派发action

    实现功能：

    1. 把一个 value 为不同 action creator 的对象，转成拥有同名 key 的对象（即把 action creators 转成拥有同名 keys 的对象）。
    2. 使用 dispatch 对每个 action creator 进行包装，以便可以直接调用它们。

    实现代码

    /**
 *  @param {*} actionCreators 是actions对象，导出的是一个个的
 *              add() {
                    return { type: TYPES.ADD };
                }
 *  @param {*} dispatch 是store提供的触发action的方法
 */
 export default function (actionCreators, dispatch){
    console.log('actionCreators',actionCreators)
    console.log('dispatch',dispatch)
    let boundActionCreators = {};
    // actionCreators 是一个对象；key是函数名，value是对应的函数体
    for (const key in actionCreators) {
        boundActionCreators[key] = function(...args){
            // dispatch方法会返回派发的action
            // 其实就是在这里派发了action，并把这个函数返回
            // ...args，参数一点要传过去，因为有可能是action带的新值
            return dispatch(actionCreators[key](...args));
        }
    }
    return boundActionCreators
 }

    export default {            //导出了一个对象，里面有两个方法：返回的是action
    add(){
        return {type: TYPES.ADD};
    },
    minus(){
        return {type: TYPES.MINUS};
    }
}
