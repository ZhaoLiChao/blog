    # redux 和 react 组件搭配基本使用

    # 1. 目录结构

    ├── src
│   ├── store                     // store，redux使用目录（redux在react中的使用）
│   │   ├── index.js              // 利用reducer调用createStore，创建store
│   │   ├── action-types.js       // 定义action的类型，共用，避免多次使用导致错误
│   │   ├── actions.js            // 多个action对象，即actionCreator
│   │   ├── reducers.js           // 处理action的方法
│   └── index.js                  //React组件

    # 2. store/action-types.js ==> 定义action的类型

    // 定义action的类型type
// 定义两个action类型，每一个type都是一个常量
export const ADD = 'ADD';
export const MINUS = 'MINUS';

    # 3. store/actions.js ==> 定义actions

    //  创建多个action，每一个action都是一个函数，返回一个对象：{type: 'XXX', payload: value}
//  这样用来创建action的函数被称为actionCreator

import * as TYPES from "./action-types";        //TYPES是一个对象，里面有两个属性：ADD、MINUS
export default {            //导出了一个对象，里面有两个方法：返回的是action
    add(){
        return {type: TYPES.ADD};
    },
    minus(){
        return {type: TYPES.MINUS};
    }
}

    # 4. store/reducers.js ==> 处理action的方法

    import * as TYPES from "./action-types";
// 定义一个初始化store的state，在调用createStore直接传一个reducer，不传初始化state
// createStore方法中会调用 dispatch({ type: '@@REDUX_INIT' }) 去调用reducer（这里）初始化state
let initialState = { number: 0 };

// 定义一个reducer方法，
/**
 * @param {*} state 初始化store中的state
 * @param {*} action 触发dispatch 动作，更加action的type（定义在action-types，共用，避免出错）改变state
 */
export default function(state = initialState, action){
    switch (action.type) {
        case TYPES.ADD:
            return {number: state.number + 1};
        case TYPES.MINUS:
            return {number: state.number - 1};
        default:
            return state
    }
}

    # 5. store/index.js ==> 调用createStore，创建store

    // store 是整个应用的仓库，需要使用redux的createStore创建

import { createStore } from "../redux";         //引入redux的createStore的方法
import reducer from "./reducer";                //引入处理action的方法

const store = createStore(reducer);         //创建store
export default store;                       //导出store给组件用

    # 6. src/index.js ==> React组件中使用store

    对于组件来说，store有以下作用：

    • 输入 把仓库中的状态在组件中显示
    • 输出 在组件里可以派发动作给仓库，从而修改仓库中的状态
    • 组件需要订阅仓库中的状态变化事件，当仓库中状态发生改变的时候要重新刷新组件
    import React from 'react';
import ReactDOM from 'react-dom';

import store from "./store";
import * as TYPES from "./store/action-types";
import actions from "./store/actions";

class Counter extends React.Component{
    constructor(prop){
        super(prop);
        this.state = {     //初始化state
            number: store.getState().number     //number默认值为store中的state
        }
    }
    componentDidMount(){
        //使用store中的createStore方法，订阅state的更新；返回的是一个取消订阅的方法
        //当仓库中的状态发生更新后，会让订阅函数执行，会更新当前组件的状态，状态更新后就会刷新组件
        this.unSubscribe = store.subscribe(() => {
            this.setState({number: store.getState().number})
        })
    }
    componentWillUnmount(){
        this.unSubscribe();         //在组件销毁的时候，取消订阅；否则会造成内存泄漏
    }
    render(){
        return(
            <div>
                <p>{this.state.number}</p>
                {/* 不使用 actions.js ，直接写action */}
                {/* <button onClick={() => store.dispatch({type: TYPES.ADD})}>+</button>
                <button onClick={() => store.dispatch({type: TYPES.MINUS})}>-</button> */}

                {/* 调用store的dispatch方法，触发actions.js里面定义的action  ==> 这是比较完整的方法，特别是在应用复制，action比较多的时候，会更省事 */}
                <button onClick={() => store.dispatch(actions.add())}>+</button>
                <button onClick={() => store.dispatch(actions.minus())}>-</button>
            </div>
        )
    }
}

ReactDOM.render(<Counter></Counter>, document.getElementById('root'));

    # 7. dispatch 的写法简化

    # 7.1 最直接，但写法累赘

    store.dispatch({type: TYPES.ADD}) ==> 直接在dispatch中写action，直接写在按钮onClick时间中

    # 7.2 使用actions.js

    store.dispatch(actions.add()) ==> actions.js => export default { add(){ return {type: TYPES.ADD} } }，类似上面的写法，在专门的一个文件里面定义类型对象

    # 7.3 在actions.js调用dispatch

    ==>即上面的写法，在专门的文件里面定义type，并且直接dispatch，页面中直接调方法即可

    // actions.js
export default {  
  add(){
    return store.dispatch({type: TYPES.ADD})
}

    // 组件中
<button onClick={() => actions.add()}>+</button>
