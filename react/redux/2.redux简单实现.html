# 1. 实现什么：

主要实现 redux 的 createStore 方法，包含三个方法：

  • getState() : 获取最新的state
  • dispatch() : 根据行为（action）调用 reducer 改变state
  • subscribe() : 订阅，将最新的 state 发送给组件

# 2. redux的简单实现

JS作用域 只有两个 一个是全局作用域，一个是函数作用域；而store中的state不能被外界访问，所以需要用函数作用域


let CHANGE_COLOR = 'CHANGE_COLOR';        // 标识要执行的动作

let initState = {color: 'red', updateCount: 0};         //store的默认初始变量

/**
 * reducer 是一个纯函数，接受一个老的状态，如果是同样的action，那么返回值肯定也是一样的
 * @param {*} oldState 旧的状态
 * @param {*} action   标识要执行行为的对象
 */
// action 的格式： {type: 'XXX', payload: value}
function reducerFn(oldState = initState, action){
    if(action.type === CHANGE_COLOR){      // 如果action的指令是 CHANGE_COLOR，则处理并返回一个新的state
        return {...oldState, color: action.payload, updateCount: oldState.updateCount + 1}
    }
    return oldState;        //否则直接返回旧的state
}

/**
 * createStore Redux应用只有一个仓库，仓库里只有一个状态State; 但会返回一个对象，包含三个方法：
 *  getState() : 获取最新的state
 *  dispatch() : 根据行为（action）改变state
 *  subscribe() : 订阅/派发，将最新的state发送给所有需要的组件
 * @param {*} reducer 
 * @param {*} initialState 
 */
function createStore (reducer,initialState){
    let state = initialState;
    let callbacks = [];         //派发的回调函数

    // 获取最新的state
    function getState(){
        return state;
    }

    // 根据行为（action）改变state
    function dispatch(action){
        // 作用：根据state和action，调用reducer，返回新的state
        state = reducer(state, action);
        callbacks.forEach(callFn => callFn());      //得到新的state后，调用所有的回调，派发新的state
    }

    // 注意：dispatch需要自己调用一次，因为有可能调用createStore时，没有传初始化的state，需要去reducer中初始化，但并不需要去修改state
    dispatch({ type: '@@REDUX_INIT' });         //不会执行reducer里面修改state的逻辑，只会初始化state

    // 订阅/派发 最新的state
    function subscribe(callback){
        callbacks.push(callback);
        // 返回一个方法，调用它后取消订阅
        // 即在组件中，订阅需要在组件销毁时取消，否则会造成内存泄露。
        return function () {        
            let index = callbacks.indexOf(callback);
            callbacks.splice(index, 1);
            //callbacks = callbacks.filter(item != callback)
        }
    }

    return {
        getState,
        dispatch,
        subscribe
    }
}

let initialState02 = {color: 'green', updateCount: 0};
let store = createStore(reducerFn, );  //没有初始化store的state，这时 dispatch({ type: '@@REDUX_INIT' }) 会调用reducer初始化state
console.log(store.getState());

store.subscribe(() => {
    console.log(store.getState());          //每次改变后打印最新的state
});
store.dispatch({ type: CHANGE_COLOR, payload: 'yellow' });
store.dispatch({ type: CHANGE_COLOR, payload: 'green' });
store.dispatch({ type: CHANGE_COLOR, payload: 'black' });
