    # Link实现

    # 1. Link语法

    # 1.1 to: string

    一个字符串形式的链接地址，通过 pathname、search 和 hash 属性创建。

    <Link to="/about">About</Link>

    # 1.2 to: object

    一个对象形式的链接地址，可以具有以下任何属性：

    • pathname - 要链接到的路径
    • search - 查询参数
    • hash - URL 中的 hash，例如 #the-hash
    • state - 存储到 location 中的额外状态数据
    <Link
  to={{
    pathname: "/courses",
    search: "?sort=name",
    hash: "#the-hash",
    state: { fromDashboard: true }
  }}
/>

    # 2. Link的实现 => Link.js

    注意：Link其实就是返回了一个a标签

    import React from "react";
import RouterContext from './RouterContext'

export default class Link extends React.Component{
    static contextType = RouterContext;
    /**
     * Link用法：<Link to="/about">About</Link>
     * this.props：{to: "/about", children: "About"}
     */
    render(){
        // Link其实就是返回了一个a标签
        // 现在的上下文对象中有两个对象，location，history
        return(
            <a {...this.props} onClick={() => {this.context.history.push(this.props.to)}}>{this.props.children}</a>
        )
    }
}

    注意：需要实现history.push方法

    # 4. 修改HashRouter.js => 主要修改render()里面的上下文对象的值

    import React from "react";
import RouterContext from './RouterContext'

/**
 * HashRouter只是一个容器，并没有DOM结构
 * 作用：渲染子组件
 *       向下传递 location 属性
 */
export default class HashRouter extends React.Component{
    state = {
        location : {
           pathname: window.location.hash.slice(1),         //获hash值，即 #/about => /about
           state: null
        }
    }
    componentDidMount(){
        // window.location.hash默认为空  ==>> 访问http://localhost:3000 时hash就为""
        window.location.hash = window.location.hash || '/';         //默认设置hash为"/"，#将会被自动加上

        // hash改变后，将会触发下面的回调函数，改变state，重新渲染组件并且理由上下文传递新的location
        window.addEventListener('hashchange', e => {
            this.setState({
                location: {
                    ...this.state.location,
                    pathname: window.location.hash.slice(1) || '/',   //更新pathname，必须去除"#"
                    state: this.locationState
                }
            })
        })
    }
    render() {
        let that = this;
        let history = {
            location: this.state.location,
            /**
             * @param {*} to 有可能是一个string ,也有可能是一个object
             * to="/about"
             * to={{
                    pathname: "/about",
                    search: "?sort=name",
                    hash: "#the-hash",
                    state: { fromDashboard: true }
                }}
             */
            push(to){       //添加一个push方法
                if(typeof to === 'object'){         //传的是一个对象，hash = pathname + search + hash（注意赋初始值，否则没传的话为undefined）
                    let {pathname, state, search = '', hash = ''} = to;
                    window.location.hash = pathname + search + hash;
                    that.locationState = state;
                }else{
                    window.location.hash = to;
                }
            }
        };
        let routerValue = {         //现在的上下文对象中有两个对象，location，history
            location: this.state.location,
            history,
        }

        return(
            // RouterContext.Provider => 使用上下文对象渲染组件，向下传递history对象
            // 下层组件使用 this.context 访问上下文对象
            <RouterContext.Provider value={routerValue}>
                {this.props.children}
            </RouterContext.Provider>
        )
    }
}

    # 6. 使用Link

    import React from 'react';
import ReactDOM from 'react-dom';
import {HashRouter as Router, Route, Link} from './react-router-dom'

import Home from './pages/home'
import About from './pages/about'
import Profile from './pages/profile'

/**
 * Router 是路由容器
 * Route 是路由对象（规则）
 * exact属性代表是否精确匹配，否则渲染'/about'和'/profile'时会同时渲染Home组件
 */
ReactDOM.render(
    <Router>
        <ul>
            <li><Link to="/">Home111</Link></li>
            <li><Link to="/about">About</Link></li>
            <li><Link to={{
                pathname: "/about",
                search: "?sort=name",
                hash: "#the-hash",
                state: { fromDashboard: true }
                }}>About</Link></li>
            <li><Link to="/profile">Profile</Link></li>
        </ul>
        <Route path='/' exact component={Home}></Route>
        <Route path='/about' component={About}></Route>
        <Route path='/profile' component={Profile}></Route>
    </Router>,
    document.getElementById('root'));
