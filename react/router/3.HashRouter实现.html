    # HashRouter的实现

    注意：
    HashRouter只是一个容器，并没有DOM结构；
    作用：渲染子组件 、向下传递 location 属性

    所以，需要定义上下文对象！！

    路由功能的实现，是由简到繁，所以后面可能有一些代码是重复的，我会把关键代码着重标签出来！

    # 1. 目录结构

    ├── react-router-dom
│   ├── RouterContext.js    // 全局上下文对象
│   ├── HashRouter.js       // HashRouter 组件实现
│   ├── Route.js            // Route 组件实现
│   └── index.js            // react-router-dom的主文件，主要用于对外导出

    # 2. RouterContext.js

    全局上下文对象在 react-router-dom 这个库中使用很频繁，所以先定义：

    // 定义一个上下文，利用它向下传递 location 属性
import React from 'react';
let context = React.createContext();
export default context;

    # 3. HashRouter.js ==>> hash路由容器

    import React from "react";
import RouterContext from './RouterContext'

/**
 * HashRouter只是一个容器，并没有DOM结构
 * 作用：渲染子组件
 *       向下传递 location 属性
 */
export default class HashRouter extends React.Component{
    state = {
        location : {
           pathname: window.location.hash.slice(1),         //获hash值，即 #/about => /about
        }
    }
    componentDidMount(){
        // window.location.hash默认为空  ==>> 访问http://localhost:3000 时hash就为""
        window.location.hash = window.location.hash || '/';         //默认设置hash为"/"，#将会被自动加上

        // hash改变后，将会触发下面的回调函数，改变state，重新渲染组件并且理由上下文传递新的location
        window.addEventListener('hashchange', e => {
            this.setState({
                location: {
                    ...this.state.location,
                    pathname: window.location.hash.slice(1) || '/',   //更新pathname，必须去除"#"
                }
            })
        })
    }
    render() {
        let routerValue = {location: this.state.location};

        return(
            // RouterContext.Provider => 使用上下文对象渲染组件，向下传递history对象
            // 下层组件使用 this.context 访问上下文对象
            <RouterContext.Provider value={routerValue}>
                {this.props.children}
            </RouterContext.Provider>
        )
    }
}

    # 4. Route.js ==>> 路由对象（规则）

    import React from "react";
import RouterContext from "./RouterContext";

export default class Route extends React.Component{
    static contextType = RouterContext;      //调用上下文   

    render(){
        /**
         * path: 路由对应的路径，可通过location.hash访问，注意 它带有"#"
         * component: 路由对应的组件
         * exact: 是否精确匹配
         */
        let {path, component: RouteComponent, exact = false} = this.props;
        let pathname = this.context.location.pathname;      //使用this.context访问上下文中的location属性

        // 注意，startsWith只能匹配到以什么开头，如匹配"/home"时也能找到"/home/xxx"，所以用在这里是不完全正确的
        // 所以，这里最好使用正则：path-to-regexp => 自行学习，这里不做记录
        if((exact && path === pathname) || (!exact && pathname.startsWith(path))){
            return(
                <RouteComponent />
            )
        }else{
            return null
        }
    }
}
