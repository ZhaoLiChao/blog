    # 前端路由基本实现

    一、HashRouter ==>> 利用hash实现

    1. HTML部分
    <body>
  <ul>
    <li><a href="#/home">Home</a></li>
    <li><a href="#/about">About</a></li>
  </ul>
  <div id="root"></div>
</body>
    1. Javascript部分
    <script>
    // 获取到渲染UI的DOM
    let root = document.getElementById('root'); 
    // hashchange 在窗口加载完成时不会监听到，所以监听一下 load
    window.addEventListener('load', onHashChange)
    // 监听 hash 的变化
    window.addEventListener('hashchange', onHashChange)

    function onHashChange() {
        const hash = window.location.hash;
        switch (hash) {
        case '#/home':
            root.innerHTML = 'Home页面';
            break;
        case '#/about':
            root.innerHTML = 'About页面';
            break;
        default:
            root.innerHTML = '首页';
            break;
        }
    }
</script>

    二、BrowserRouter ==>> 利用history实现

    1. HTML部分
    <body>
   <div id="root"></div>
</body>
    1. Javascript部分
    <!-- history 路由的基本实现 -->
  <script>
    // 注意：history 路由也可以参照hash路由的实现方式，但是a标签本身就带有跳转功能，需要拦截哦
    // 这里，换一种简单的方式实现
    let root = document.getElementById('root'); 

    // ========================= 监听onpushstate（添加路由） =========================
    window.onpushstate = function(state, title, url){
      console.log('===pushstate',state, title, url)
      root.innerHTML = url;
    }
    // window本身没有onpushstate，需要自己实现
    ;(function(history){
      let oldPushState = history.pushState;     //先保存下原本旧的pushstate方法
      /** 
        * pushSate和popState 都有三个常用参数：
        * state：路由状态；
        * title：路由标题；
        * url：路由目标路径
      */
      history.pushState = function(state, title, url){  //重写pushState方法
        oldPushState.call(history, state, title, url);    //先调用旧的pushstate方法
        //判断一下自己是否在当前这个window环境下（不是当前这个IIFE哈）定义onpushstate
        if(typeof window.onpushstate === 'function'){     
          window.onpushstate(state, title, url);
        }
      }
    })(window.history);      //window.history作为参数

    // ========================= 监听onpopstate（删除路由） =========================
    window.onpopstate = function(e){
      console.log('====popstate',e)
      // e 为onpopstate的事件源，比较重要的三个属性：type,state,window.location.pathname
      let mianInfo = {
        type: e.type,                       //type 为事件类型
        state: e.state,                     //state 为当前路由的状态，即 {page: 'home'}
        pathName: window.location.pathname  //当前路由的路径，即 '/home'
      }
      root.innerHTML = mianInfo.pathName;
    }

    // ========================= 测试效果代码 =========================
    setTimeout(() => {
      window.history.pushState({page: 'home'}, 'title 1', '/home')
    }, 2000)

    setTimeout(() => {
      window.history.pushState({page: 'about'}, 'title 2', '/about')
    }, 4000)

    setTimeout(() => {
      window.history.pushState({page: 'profile'}, 'title 3', '/profile')
    }, 6000)

    setTimeout(() => {
      window.history.go(-1)
    }, 8000)
    
  </script>

    三、总结

    • 调用history.pushState()或history.replaceState()不会触发popstate事件
    • 浏览器动作时，才会触发该事件，如用户点击浏览器的回退按钮（或者在Javascript代码中调用history.go()或者history.back()或者history.forward()方法）
    • 在BrowserRouter模式下，收到修改hash不会触发onpushstate事件，但是会触发onpopstate事件
