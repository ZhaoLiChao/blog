    # Switch 和 Redirect

    # 一、Switch组件

    # 1. 作用

    该组件下面是多个Route（可能还会有一个 Redirect ），Switch保证只渲染其中的一个

    # 2. 实现

    import React from "react";
import {pathToRegexp} from 'path-to-regexp';
import RouterContext from "./RouterContext";

export default class Switch extends React.Component{
    static contextType  = RouterContext;        //调用上下文

    render(){
        let pathname = this.context.location.pathname;      //使用this.context访问上下文中的location属性，获取当前路径
        let childrens = this.props.children;                //获取Switch下面的所有孩子节点
        childrens = Array.isArray(childrens) ? childrens : [childrens];     //组成数组

        // 遍历孩子节点数组，判断是否匹配
        for (let i = 0; i < childrens.length; i++) {
            let child = childrens[i];
            let {path = "/", component, exact = false} = child.props;

            // 根据孩子节点的path和当前路径pathname做对比（注意exact：是否精确匹配）
            let regexp = pathToRegexp(path, [], { end: exact });        //匹配完全符合pathname的组件
            let matched = pathname.match(regexp);
            if(matched){
                return child;       //因为Switch组件下面一般为Route组件，所以需要返回child而不是component
            }
        }
    }
}

    # 3. 使用

    <Switch>
    <Route path="/" exact component={Home} />
    <Route path="/user" component={User} />
    <Route path="/profile" component={Profile} />
    <Redirect to="/" />
</Switch>

    # 二、Redirect组件

    # 1. 作用

    导航到新位置，即重定向 主要参数：to：要重定向组件的路由地址，即Route中定义的 path 参数。
    一般Redirect都放在Switch里面，并且必须放在Switch的最后一行。
    作用：如果上面的路由都匹配不到是，就跳转到Redirect对应的页面

    另一个参数：from ==>> 由哪个路径进行重定向，即路径为from时，将重定向到to。
    如：<Redirect from="/home" to="/login" /> ==>> 表示路径为/home时，将重定向到/login

    # 2. 实现

    import React from "react";
import RouterContext from "./RouterContext";

export default class Redirect extends React.Component{
    static contextType = RouterContext;      //调用上下文
    render(){
        let pathname = this.context.location.pathname;      //使用this.context访问上下文中的location属性

        /**
         * Redirect 组件的两种用法
         * <Redirect to='/profile' />
         * <Redirect from="/home" to="/profile" />
         */
        // 如果没有 from ，或者 from === pathname（当前路径）；跳转到 to 属性
        if(!this.props.from || this.props.from === pathname){
            this.context.history.push(this.props.to);
        }
        return null;      //否则不做任何处理
    }
}

    # 3. 使用

    <Switch>
    <Route path='/' exact component={Home}></Route>
    <Route path='/about' component={About}></Route>
    <Route path='/profile' component={Profile}></Route>
    {/* <Redirect to='/about' /> */}
    <Redirect from="/home" to="/about" />
</Switch>
