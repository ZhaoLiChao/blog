# 函数节流与防抖

# 一、概念

# 1. 为什么使用防抖和节流

对于频繁触发的事件 比如keydown、keyup、resize、scroll、mousemove事件，当频繁点击时候，会多次触发事件
导致：页面出现卡顿，影响性能
使用目的：提升性能，提高用户体验

# 2. 它们之间的区别

函数防抖（debounce）：间隔时间内只执行一次 ==>> 就是指触发事件后在 n 秒内函数只能执行一次，如果在 n 秒内又触发了事件，则会重新计算函数执行时间。 即在你结束操作后 n 秒内执行一次

函数节流（throttle）：间隔时间内执行 ==>> 就是指连续触发事件但是在 n 秒中只执行一次函数。节流会稀释函数的执行频率。 即每隔 n 秒执行一次

# 3.使用场景

函数防抖：搜索框等
函数节流：页面滚动、鼠标不断点击事件等

# 二、函数防抖的实现

我们以 onmousemove 方法为例解释防抖函数

防抖的作用：指触发事件后在 n 秒内函数只能执行一次，如果在 n 秒内又触发了事件，则会重新计算函数执行时间

# 1. 先来一个基本的onmousemove效果

<div id="content" style="height:150px;line-height:150px;text-align:center; color: #fff;background-color:#ccc;font-size:80px;"></div>
<script>
    let num = 1;
    let content = document.getElementById('content');

    function showNum() {
        content.innerHTML = num++;
    };
    // 只要在盒子内部一定，数字都将加1  ==>> 如果在盒子内部快速移动鼠标，则会反复被执行
    content.onmousemove = showNum;
</script>

# 2. 给onmousemove加上一个防抖函数

<script>
        let content = document.getElementById('content')
        let num = 1
        function showNum() {
            content.innerHTML = num++
        }
        /** 
            debounce: 防抖函数，回调函数延迟规定时间后执行，如果在规定时间内被再次调用，则重新计时，返回一个函数
            func: 延迟执行的回调函数
            awaitTime: 延迟执行的时间
        */
        function debounce(func, awaitTime) {
            let timer
            return function () {        // 必须返回一个函数，保证每一次都会被重新执行，否则只能执行一次
                let context = this              // this将是调用防抖函数的DOM, 即 content
                let args = arguments        // 保存参数（这里没有）

                // 如果定时器已经存在，则表示在规定时间内重复调用了
                if(timer) {
                    clearTimeout(timer)
                }
                // 在规定时间后调用回调函数
                timer = setTimeout(() => {
                    func.apply(context, args)
                }, awaitTime)
            }
        }
        // 1s内只执行一次
        content.onmousemove = debounce(showNum, 1000)
    </script>

效果(加上防抖)：

注意：这样加上防抖有一个问题，鼠标移入的时候内容没有立即显示。

# 2. 给onmousemove加上一个防抖函数（立即执行版）

function debounce(func, awaitTime) {
    let timer
    return function () {        // 必须返回一个函数，保证每一次都会被重新执行，否则只能执行一次
        let context = this              // this将是调用防抖函数的DOM, 即 content
        let args = arguments        // 保存参数（这里没有）

        // 如果定时器已经存在，则表示在规定时间内重复调用了
        if(timer) {
            clearTimeout(timer)
        }
        let callNow = !timer
        // 在规定时间后调用回调函数
        timer = setTimeout(() => {
            // timer = null             // 在规定时间后清除定时器，下一次调用 callNow 将是true，会改变内容，然后保存下一个timer
            func.apply(context, args)   // 如果是这种写法：则callNow只能在第一次的时候有用
        }, awaitTime)
        
        // 立即执行一次
        if(callNow){
            func.apply(context, args)
        }
    }
}
// 给dom绑定mousemove，只要在dom内移动鼠标，都将触发回调函数 1s内只执行一次
content.onmousemove = debounce(showNum, 1000)

# 3. 合并非立即执行和立即执行版本（加一个参数在调用处判断）

/** 
    debounce: 防抖函数，回调函数延迟规定时间后执行，如果在规定时间内被再次调用，则重新计时，返回一个函数
    func: 延迟执行的回调函数
    awaitTime: 延迟执行的时间
    immediate: 是否立即执行, 默认值为false
    */
    function debounce(func, awaitTime, immediate = false) {
    let timer
    return function() {
        let context = this
        let args = arguments

        if(timer)  clearTimeout(timer)
        if(immediate){      // 立即执行版本
            let callNow = !timer
            timer = setTimeout(() => {
                timer = null;
            }, awaitTime)
            if (callNow) func.apply(context, args)

        } else {             // 非立即执行版本
            timer = setTimeout(function(){
                func.apply(context, args)
            }, awaitTime);
        }
    }
}

# 二、函数节流的实现

还是以 onmousemove 方法为例解释防抖函数

节流的作用：连续触发事件但是在 n 秒中只执行一次函数。节流会稀释函数的执行频率

# 1. 利用时间戳

<script>
    let content = document.getElementById('content')
    let num = 1
    function showNum() {
        content.innerHTML = num++
    }
    
    function throttle(func, awaitTime){
        let previous = 0     // 上一次调用的毫秒数
        return function() {
            let now = Date.now()    // 这一次调用的毫秒数
            let context = this
            let args = arguments
            if(now - previous > awaitTime) {    // 这一次调用时间 - 上一次的调用时间 > 规定的价格时间
                func.apply(context, args)
                previous = now      // 将这一次的时间保存
            }
        }
    }
    // 每隔 1s 数组加1
    content.onmousemove = throttle(showNum, 1000)
</script>

# 2. 利用定时器

<script>
    let content = document.getElementById('content')
    let num = 1
    function showNum() {
        content.innerHTML = num++
    }
    
    function throttle(func, awaitTime){
        let timer
        return function() {
            let context = this
            let args = arguments
            if(!timer){     // 定时器不存在
                timer = setTimeout(() => {
                    // 在规定时间后清除定时器，但是节流一般调用它的函数都在快速的重复执行，所以会定义一个新的定时器，再次执行
                    timer = null
                    func.apply(context, args)       // 执行回调
                }, awaitTime)
            }
        }
    }
    content.onmousemove = throttle(showNum, 1000)
</script>

效果和时间戳版是一致的，只是两中不同的实现方式。如果有需要，可以加一个参数，将两种方式合并到一起。（防抖函数中有写）
