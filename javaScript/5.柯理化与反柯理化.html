# 柯理化与反柯理化

# 一、柯理化

# 1. 是什么？

函数柯里化就是把接受多个参数的函数变换成接受一个单一参数的函数，并且返回接受余下参数返回结果的技术

# 2. 实现柯理化

# 2.1. 想要什么结果？

// 实现函数 add , 使下面的输出结果一致
console.log(add(1, 2, 3, 4, 5));//15
console.log(add(1)(2)(3)(4)(5));//15
console.log(add(1, 2)(3, 4, 5));//15

# 2.2. 实现方式一（根据想要的结果写一个递归函数，有很大局限性）

let add = (function (argLength){
    let allArgs = [];       // 所有的参数数组
    function _add(...args) {
        allArgs = [...allArgs, ...args];        // 每次都将参数放到数组里面
        if(allArgs.length >= argLength) {       //如果参数列表里面已经有想要的那么多个参数，则处理
            // 利用数组的reduce方法将所有参数相加
            let result = allArgs.reduce((prev, current) => prev + current, 0);
            allArgs = [];    // 置空数组
            return result
        } else {        // 否则将继续收集参数
            return _add
        }
    }
    return _add
})(5)

# 2.3. 实现方式二

function add(...args) {
    /**
     * 每次调用都构建新一个新的函数 并返回
     * 这个新函数将这次调用的 所有参数合并的以前的上面，使用bind， 传入了一个null, 其实相当于没有改变this，只是在收集参数
     */
    let _add = add.bind(null, ...args);     // 每一次
    // 使用alert的时候，会调用toString，这个改写他的toString方法，实现自己想要的结果
    _add.toString = function () {
        return args.reduce((previousValue, currentValue) => previousValue + currentValue, 0);
    }
    return _add;            // 首先，每一次都返回一个函数
}

// ========== 测试代码  toString只能使用alert ==========
alert(add(1, 2, 3, 4, 5));
alert(add(1)(2)(3)(4)(5));
alert(add(1, 2)(3, 4, 5));

# 4. 实现方式三，使用柯理化函数

// fn 代表条件满足之后的处理函数，其参数代表接受多少个参数
function curry(fn, ...args) {
    // 果然当前函数的参数个数已经满足 fn 的参数个数，则处理，否则继续递归收集参数
    return args.length < fn.length ? (...innerArgs) => {
        // 添加满足后，调用fn处理结果
        return curry(fn, ...args, ...innerArgs);
    } : fn(...args);
}

// 函数的length属性代表函数形参的个数
function addFn(a, b, c, d, e) {
    return a + b + c + d + e;
}
let add = curry(addFn);
console.log(add(1, 2, 3, 4, 5));//15
console.log(add(1)(2)(3)(4)(5));//15
console.log(add(1, 2)(3, 4, 5));//15

# 二、反柯理化

# 1. 简单的反柯理化

Function.prototype.uncurring = function () {
    let self = this        // 保存调用uncurring的函数，即this
    return function () {   // uncurring 将返回一个新的函数
        /**
         * 函数调用uncurring后，将返回一个新的函数，使用参数将在这里接受
         * 注意：
         *     1. arguments 是一个类数组，如：{ '0': 1, '1': 2, '2': 3 }
         *     2. 如果需要将 arguments 转化为数组，let args = Array.prototype.slice.call(arguments);
         */

        /**
         * 运行过程
         * push.call({'0':1, '1':2, '2':3},[4,5]);
         * 即 {'0':1, '1':2, '2':3}.push(4,5)
         */
        let obj = Array.prototype.shift.call(arguments);
        return self.apply(obj, arguments)
    }
}
let push = Array.prototype.push.uncurring();
(function () {
    push(arguments, 4, 5);
    //Array.prototype.push.call(arguments,4,5);
    console.log(arguments); // { '0': 1, '1': 2, '2': 3, '3': 4, '4': 5 }
})(1, 2, 3);
