# 组件通信的方式

快速原型开发: 可以快速识别.vue文件封装组件插件等功能,也是基于Vue-Cli

npm install @vue/cli -g
npm install -g @vue/cli-service-global

# 一、props & $emit

# 1. 父组件

<template>
    <!-- 组件通信 ===>> 父组件 -->
    <div class="outBox">
        <p>父组件</p>
        <!-- 3. 使用子组件 -->
        <son :name='name' @handleClick='handleClick'></son>
    </div>
</template>

<script>
import Son from './son'         // 1.引入子组件
export default {
    data(){
        return {
            name: 'zyc'
        }
    },
    components: {
        Son         // 2. 注册子组件
    },
    mounted(){
        setTimeout(() => {
            // name 更新后会同步更新到子组件中
            this.name = 'zhaoyichao'
        }, 5000)
    },
    methods: {
        handleClick(value) {     // value 就是子组件在$emit时传回来的参数，可以有多个
            console.log('====value', value)
        }
    }
}
</script>

<style lang="stylus" scoped>
.outBox{
    & > p{
        border-bottom solid 1px #ccc
        margin-bottom 30px
    }
}
</style>

# 2. 子组件

<template>
    <!-- 组件通信 ===>> 子组件 -->
    <div>
        子组件 {{name}} <br />
        <button @click="handleClick">$emit向上触发按钮</button>
    </div>
</template>

<script>
export default {
    props: {
        name: {   // 定义父组件需要传递的参数名，类型和默认值
            type: String,
            default: ''
        }
    },
    methods: {
        handleClick() {
            this.$emit('handleClick', 'sonName')
        }
    }
}
</script>

# 3. $emit 同步父子组件的数据,==>>语法糖的写法

  • 修改子组件方法
handleClick() {
    // this.$emit('handleClick', 'sonName')

    // 固定写法 update:要更新的属性名
    this.$emit('update:name', '100')        // 将自动更新父组件中 使用.sync修饰符绑定并传递到子组件的值
}
  • 父组件中传递属性时，使用sync
<!-- 子组件中调用了 this.$emit('update:name', '100')。将自动更新name的值 -->
<son :name.sync='name'></son>

# 4. $emit 同步父子组件的数据 ==>> 更简单的写法

  • 修改子组件
// 子组件接受的props必须定义为 value
handleClick() {
    this.$emit('input', '100')
}
  • 父组件必须使用 v-model 传递参数，不要定义自定义事件，默认触发 update:input 事件
<!-- 3. 使用子组件 -->
<son v-model='name'></son>

// 父亲组件index.vue中使用了 儿子2组件son2.vue；并且自定义了事件 @say

# 导致的问题

如果层级很深那么就会出现 $parent.$parent..... 或者 $children.$children..... 这种层级很深的情况！！不好理解。所以我们可以在main.js中封装共用方法：
封装一个$dispatch方法向上进行派发，即调用$parent
封装一个$broadcast方法向下进行派发，即调用$children

  • $dispatch
// 儿子找到爸爸 让爸爸方法触发
Vue.prototype.$dispatch = function (eventName,value) {
    let parent = this.$parent; // 先找第一层的$parent
    while (parent) {  //如果找到了$parent
        parent.$emit(eventName, value); // 触发方法，如果没有方法将被忽略
        parent = parent.$parent; // 接着向上找
    }
}
  • $broadcast
// 爸爸找到儿子 让儿子对应的方法触发
Vue.prototype.$broadcast = function (eventName,value) {
   let children = this.$children;       // 先找第一层的$children，返回的是数据
    function broad(children) {
        children.forEach(child => { // 如果自己的儿子下面 还有儿子 继续查找
            child.$emit(eventName, value); // 触发当前儿子上的对应事件
            if (child.$children){       // 然后还有下及，继续递归查找
                broad(child.$children)
            }
        });
    }
    broad(children); // 先找自己的儿子
}

使用封装的$dispatch & $broadcast

  • 父组件中使用 $broadcast
handleParent() {
    // 在父组件中点击，$emit触发了子组件son2自定义的 say 方法
    this.$broadcast('say', 'parent')
},
  • 子组件中使用 $dispatch
handleParent() {
    // 在父组件中点击，$emit触发了子组件son2自定义的 say 方法
    this.$dispatch('say', '11111')
},

# 三、$attrs & $listeners

# $attrs： 属性的集合

# $listeners： 方法的集合

使用方法：v-bind 将属性全部向下传递 v-on 将方法全部进行向下传递
即 将父组件传递给 子组件 的所有属性或者方法，传递给孙子组件

注意：

  • 如果组件中使用了 props 就会将 attrs 从当前$attrs移除掉
  • 但是可以在子组件是使用 / @click="$listeners.xxx" 访问属性或者方法 并且 继续往下传递
  • 在要使用的组件中不用定义props接收 $attrs

inheritAttrs ：是否在标签上面显示自定义属性，默认为true，即会显示。可设为false

实例代码：

# 1. 父组件

<template>
    <!-- 组件通信 ===>> 父组件 -->
    <div class="outBox">
        <p>父组件</p>
        <son2 :name="name" :count="count" @handleClick="handleClick" @handleMouseup="handleMouseup"></son2>
    </div>
</template>

<script>
import Son2 from './son2'         // 1.引入子组件
export default {
    data(){
        return {
            name: 'zyc',
            count: 10
        }
    },
    components: {
        Son2
    },
    mounted(){
       
    },
    methods: {
        handleClick() {
            console.log('在父组件中给 子组件son2 自定义的click事件')
        },
        handleMouseup(){
            console.log('在父组件中给 子组件son2 自定义的Mouseup事件')
        }
    }
}
</script>

<style lang="stylus" scoped>
.outBox{
    & > p{
        border-bottom solid 1px #ccc
        margin-bottom 30px
    }
}
</style>

# 2. 子组件

<template>
    <!-- 组件通信 ===>> 子组件 -->
    <div>
        <!-- 使用$listeners中的某一个方法 -->
        <p @click="$listeners.handleClick">子组件22222</p>
        <!-- 使用$attrs中的某一个属性 -->
        {{$attrs.count}}
        <!-- v-bind  将属性全部向下传递   v-on 将方法全部进行向下传递-->
        <grandson v-bind="$attrs" v-on="$listeners"></grandson>
    </div>
</template>

<script>
import Grandson from './grandson'
export default {
    // 不能使用 props，否则就会将 attrs 从当前$attrs移除掉
    inheritAttrs: false,   // 设置不在标签上面显示自定义属性 name / count
    props: {
        // name:{
        //     type: String
        // }
    },
    components: {
        Grandson
    },
    methods: {
    }
}
</script>

# 3. 孙子组件

<template>
    <div>
        <!-- 使用 $attrs 和 $listeners -->
        <p @click="$listeners.handleClick">grandson</p>
        {{$attrs.name}}
        <button @click="handle" @mouseup="$listeners.handleMouseup">grandson按钮</button>
    </div>
</template>

# 四、Provide & Inject

# 1. Provide：提供全局数据

<script>
import Son2 from './son2'         // 1.引入子组件
export default {
    provide(){ // 暴露全局方法
        return {
            vm: this,       // 暴露整个组件实例         
        }
    },
    data(){
        return {
            name: 'zyc',
            count: 10
        }
    },
    components: {
        Son2
    },
    mounted(){
    },
    methods: {
        say(str){
            console.log('======父组件的say方法,', str)
        }
    }
}
</script>

# 2. Inject：在任意子组件中注入父级数据

<script>
export default {
    // 如果注入的变量名 找不到，则会报错

    // vm 是父组件提供，sonMsg 是子组件提供数据
    inject:['vm', 'sonMsg'], // 注入 会像上查找
    mounted(){
        // this.vm 可以访问到整个父组件实例
        console.log('========this.vm', this.vm.name)
        console.log('========this.sonMsg', this.sonMsg)
        this.vm.say('在孙子组件中调用')
    },
    methods: {
    }
}
</script>

注意：

  • provide & inject 在实例中都是和data/mounted同级
  • 在子组件中注入了父组件的数据后，就变成了子组件的全局变量，使用 this.xxx 访问
  • 通过方法回调实现数据通信：在子组件中访问 this.xxx.方法名(value) 并传参，在父组件中就可以接收，实现数据通信
  • 可以在某个子组件中注入(inject)，同时也可以提供(provide) 数据

# 五、Ref使用

获取组件实例，ref被用来给元素或子组件注册引用信息。引用信息将会注册在父组件的 $refs 对象上。
即只能获取子组件的实例，不能在子组件中通过ref获取父组件实例

# 1. 子组件中定义数据和方法

<script>
import Grandson from './grandson'
export default {
    props: {
    },
    data(){
        return{
            name: 'child'
        }
    },
    mounted(){
    },
    components: {
        Grandson
    },
    methods: {
        say(){
            console.log('======子组件中say')
        }
    }
}
</script>

# 2. 父组件中通过refs访问组件实例

<template>
    <!-- 组件通信 ===>> 父组件 -->
    <div class="outBox">
        <p>父组件</p>
        <son2 ref="son2"></son2>
    </div>
</template>

<script>
import Son2 from './son2'         // 1.引入子组件
export default {
    data(){
        return {
            name: 'zyc',
            count: 10
        }
    },
    components: {
        Son2
    },
    mounted(){
        let son2Vm = this.$refs.son2
        console.log('=====son2Vm', son2Vm)
        // 访问子组件的数据
        console.log('======son2Vm.name', son2Vm.name)
        // 调用子组件的方法
        son2Vm.say()
    },
    methods: {
    }
}
</script>

# 六、EventBus

用于跨组件通信(不复杂的项目可以使用这种方式)
原理：就是创建一个全局的发布订阅 vm.$on $emit

注意：可以在任意组件间通信
即：可以在父组件中定义事件，子组件中触发；也可以在子组件中定义事件，父组件中触发。

  • 父组件中定义事件（发布）
mounted(){
    // 利用发布定义  自定义一个customSay 方法，接受一个参数
    this.$bus.$on('customSay', data => {
        console.log('=say:', data)
    })
},
  • 孙子组件中触发事件（订阅）
mounted() {
    // 孙子组件的mounted被调用时，父组件的mounted还没有调用执行，所以需要放置$nextTick中
    this.$nextTick(() => {
        this.$bus.$emit('customSay', '孙子组件')
    })
}
